CREATE TABLE IF NOT EXISTS vouchers (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    crew_name TEXT NOT NULL,
    crew_id TEXT NOT NULL,
    flight_number TEXT NOT NULL,
    flight_date TEXT NOT NULL,
    aircraft_type TEXT NOT NULL,
    seat1 TEXT NOT NULL,
    seat2 TEXT NOT NULL,
    seat3 TEXT NOT NULL,
    created_at TEXT NOT NULL DEFAULT (datetime('now'))
);

CREATE TABLE IF NOT EXISTS users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    username varchar(255) NOT NULL UNIQUE,
    email varchar(255) UNIQUE NULL,
    privilege varchar(255) NOT NULL,
    password_hash varchar(255) NOT NULL,
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    updated_at TEXT NOT NULL DEFAULT (datetime('now'))
);

CREATE INDEX IF NOT EXISTS idx_vouchers_crew_id
    ON vouchers (crew_id);

CREATE INDEX IF NOT EXISTS idx_vouchers_flight
    ON vouchers (flight_number, flight_date);

CREATE INDEX IF NOT EXISTS idx_vouchers_created_at
    ON vouchers (created_at);
